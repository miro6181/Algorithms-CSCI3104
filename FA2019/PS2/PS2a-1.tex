\item{\itshape (6 pts) For each of the following pairs of functions $f(n)$ and $g(n)$, we have that $f(n) \in \mathcal{O}(g(n))$. Find valid constants $c$ and $n_{0}$ in accordance with the definition of Big-O. For the sake of this assignment, both $c$ and $n_{0}$ should be strictly less than $10$. You do \textbf{not} need to formally prove that $f(n) \in \mathcal{O}(g(n))$ (that is, no induction proof or use of limits is needed).}
\begin{enumerate}[label=(\alph*)]
\item $f(n) = n^{3}\log(n)$ and $g(n) = n^{4}$.
\makenonemptybox{1.5in}{$f(n) = n^{3}log(n)$, $g(n) = n^{4}$ $\Rightarrow$ $n^{3}log(n) \in O(n^{4})$ \\$\Rightarrow$ $n^{3}log(n) \leq C * n^{4}$ for all $n \geq n_0$ \\for $c=2$ and $n_0=3$ : \\ $n^{3}log(n) \leq C * n^{4}$ = $\frac{3^{2}log(3)}{3^{4}} \leq 2$ $\Rightarrow$ $\frac{log(3)}{3} \leq 2$ $\Rightarrow$ $0.159 \leq 2$ \\$\therefore$ $\frac{n^{3}log(n)}{n^{4}} \leq c$ for all $n \geq n_0$ where $c=2$ and $n_0 = 3$}

\item $f(n) = n2^{n}$ and $g(n) = 2^{n\log_{2}(n)}$.
\makenonemptybox{1.5in}{$n2^{n} \leq C*2^{nlog_2(n)}$ per the definition of Big-O \\ For $C=3$ and $n_0=2$ : \\ $\frac{2^{n}n}{2^{nlog_2(n)}} \leq C$ $\Rightarrow$ $\frac{2^{2}2}{2^{2log_2(2)}} \leq 3$ $\Rightarrow$ $2 \leq 3$ \\ \\Becuase the denomiator is growing larger than the numerator, the term on the left approaches zero, therefore $C=3$ and $n_0 = 2$ hold for $n \geq n_0$} 


\item $f(n) = 4^{n}$ and $g(n) = (2n)!$
\makenonemptybox{1.5in}{$4^{n} \leq C * (2n)!$ per the definition of Big-O $\Rightarrow$ $\frac{4^{n}}{(2n)!} \leq C$ \\ For $C=2$, $n_0=1$: \\ $\frac{4^{1}}{(2(1)!} \leq 2$ $\Rightarrow$ $2\leq 2$ \\ This holds for all $n\geq n_0$ because the factorial term grows far faster than the linear term which makes the LHS arrpoach zero. \\ $\therefore$ $C=2$, $n_0 = 1$}
\end{enumerate}

\item {\itshape (2 pts) Consider the following algorithm. Find a suitable function $g(n)$, such that the algorithm's worst-case runtime complexity is $\Theta(g(n)).$ You do \textbf{not} need to formally prove that $f(n) \in \Theta(g(n))$ (that is, no induction proof or use of limits is needed).}
\begin{verbatim}
count = 0
for(i = n; i >= 0; i = i - 1){
    for(j = i-1; j >= 0; j = j-1){
        count = count+1
    }
}
\end{verbatim}
\makenonemptybox{5in}{$T(n) = C_1 + C_2(n+1) + C_3(n)+C_4(n)$}
