	FindElement(A, n) : //suppose no duplicates in array A and array A is not empty

	    for i = 0 to length(A)-1 {
	        if A[i] == n{
	            ret = i	           
	    }}
	    return ret
	\end{verbatim}
	\end{small}
\makenonemptybox{2.5in}{ret stores either $-1$ or i at the start of the i\textsuperscript{th} iteration.}

	SumArray(A) : //suppose array A is not empty

	        sum += A[i]    
	    }
	    return sum
	\end{verbatim}
	\end{small}
\makenonemptybox{2.5in}{sum is storing A[0]+A[1]+A[2]+ ... + A[i] at the i\textsuperscript{th} iteration.}
